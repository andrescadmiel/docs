---
title: "Create User"
api: "POST https://api.wikichat.com/v1/users"
description: "Create a new user in the system"
---

Creates a new user with the provided information. The user will be assigned a unique ID and default settings.

## Body Parameters

<ParamField body="name" type="string" required>
  The user's display name (3-50 characters)
</ParamField>

<ParamField body="email" type="string" required>
  The user's email address (must be unique)
</ParamField>

<ParamField body="password" type="string" required>
  The user's password (minimum 8 characters)
</ParamField>

<ParamField body="profile" type="object">
  Optional profile information
  
  <Expandable title="profile">
    <ParamField body="bio" type="string">
      User bio (max 500 characters)
    </ParamField>
    
    <ParamField body="location" type="string">
      User's location
    </ParamField>
    
    <ParamField body="website" type="string">
      User's website URL
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="preferences" type="object">
  User preferences
  
  <Expandable title="preferences">
    <ParamField body="notifications_enabled" type="boolean" default="true">
      Whether to enable email notifications
    </ParamField>
    
    <ParamField body="theme" type="string" default="light">
      UI theme preference (`light`, `dark`, or `auto`)
    </ParamField>
    
    <ParamField body="timezone" type="string" default="UTC">
      User's timezone (IANA format)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="id" type="string">
  The newly created user's unique identifier
</ResponseField>

<ResponseField name="name" type="string">
  The user's display name
</ResponseField>

<ResponseField name="email" type="string">
  The user's email address
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the user was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of when the user was last updated
</ResponseField>

<ResponseField name="profile" type="object">
  User profile information
  
  <Expandable title="profile">
    <ResponseField name="bio" type="string">
      User bio
    </ResponseField>
    
    <ResponseField name="location" type="string">
      User's location
    </ResponseField>
    
    <ResponseField name="website" type="string">
      User's website URL
    </ResponseField>
    
    <ResponseField name="avatar_url" type="string">
      Default avatar URL (generated)
    </ResponseField>
  </Expandable>
</ResponseField>

```bash cURL
    },

      "theme": "dark",
      "timezone": "America/New_York"
    }
  }'
```

```python Python
import requests

response = requests.post(
    'https://api.wikichat.com/v1/users',
    headers={
        'Authorization': 'Bearer YOUR_API_KEY',
        'Content-Type': 'application/json'
    },
    json={
        'name': 'Jane Smith',
        'email': 'jane@example.com',
        'password': 'securepassword123',
        'profile': {
            'bio': 'Product manager passionate about user experience',
            'location': 'New York, NY'
        },
        'preferences': {
            'theme': 'dark',
            'timezone': 'America/New_York'
        }
    }
)

user = response.json()
print(user)
```
